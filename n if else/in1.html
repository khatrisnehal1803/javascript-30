function checkVotingEligibility(age, citizenship) { if (age >= 18) { // Outer if condition to check age if (citizenship === 'citizen') { // Nested if condition to check citizenship console.log("You are eligible to vote."); } else { console.log("You must be a citizen to vote."); } } else { if (age > 0) { console.log("You are not old enough to vote."); } else { console.log("Invalid age provided."); } } }
